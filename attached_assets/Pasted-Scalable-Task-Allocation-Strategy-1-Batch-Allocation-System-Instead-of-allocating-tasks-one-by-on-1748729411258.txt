async function allocateTaskBatch(batchId: number) {
  const batch = await getBatch(batchId);
  const eligibleUsers = await findEligibleUsers(batch.allocation_criteria);
  const selectedUsers = await selectUsers(eligibleUsers, batch.target_user_count);
  
  // Batch insert for performance
